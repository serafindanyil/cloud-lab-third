import os
import json
import boto3

sqs = boto3.client('sqs')
QUEUE_URL = os.environ['QUEUE']

def lambda_handler(event, context):
    print("Received:", event)

    response = sqs.send_message(
        QueueUrl=QUEUE_URL,
        MessageBody=json.dumps(event)
    )

    return {
        "statusCode": 200,
        "body": json.dumps({
            "status": "OK",
            "messageId": response["MessageId"]
        })
    }
